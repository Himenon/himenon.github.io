## Alias


```conf
[alias]
	st = status
	bl = blame
	cm = commit
	ch = checkout
	br = branch
[filter "lfs"]
	clean = git-lfs clean -- %f
	smudge = git-lfs smudge -- %f
	process = git-lfs filter-process
	required = true
[credential]
	helper = manager
```
